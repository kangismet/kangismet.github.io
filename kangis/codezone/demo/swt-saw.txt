<script>
(function () {
  const SELECTOR = '.post-body p, .post-body li, .post-body span, .comments p';
  const rules = [
    {
      regex: /\b(Muhammad|Nabi Muhammad|Rasulullah)\s*(?:\((SAW|saw)\)|\b(SAW|saw)\b)/g,
      replace: '$1 <span class="lafadz-saw" aria-label="Shallallahu alaihi wasallam">ﷺ</span>'
    },
    {
      regex: /\b(Allah)\s*(?:\((swt|SWT)\)|(swt|SWT))\b/g,
      replace: '$1 <span class="lafadz-swt" aria-label="Subhanahu wa ta\'ala">ﷻ</span>'
    },
    {
      regex: /سُبْحَانَهُ وَتَعَالَى/g,
      replace: '<span class="lafadz-swt" aria-label="Subhanahu wa ta\'ala">ﷻ</span>'
    },
    {
      regex: /\((as)\)/gi,
      replace: '<span class="arabic-inline" aria-label="Alaihis salam">عَلَيْهِ السَّلَامُ</span>'
    },
    {
      regex: /\((ra)\)/gi,
      replace: '<span class="arabic-inline" aria-label="Radhiyallahu anhu">رَضِيَ اللهُ عَنْهُ</span>'
    },
    {
      regex: /\((radhiyallahu\s+anha)\)/gi,
      replace: '<span class="arabic-inline" aria-label="Radhiyallahu anha">رَضِيَ اللهُ عَنْهَا</span>'
    },
    {
      regex: /\((radhiyallahu\s+anhuma)\)/gi,
      replace: '<span class="arabic-inline" aria-label="Radhiyallahu anhuma">رَضِيَ اللهُ عَنْهُمَا</span>'
    },
    {
      regex: /\((radhiyallahu\s+anhum)\)/gi,
      replace: '<span class="arabic-inline" aria-label="Radhiyallahu anhum">رَضِيَ اللهُ عَنْهُمْ</span>'
    },
    {
      regex: /\((rahimahullah)\)/gi,
      replace: '<span class="arabic-inline" aria-label="Rahimahullah">رَحِمَهُ اللهُ</span>'
    }
  ];
  document.querySelectorAll(SELECTOR).forEach(el => {
    if (el.closest('pre, code')) return;
    let html = el.innerHTML;
    rules.forEach(rule => {
      html = html.replace(rule.regex, rule.replace);
    });
    el.innerHTML = html;
  });
})();
</script>
